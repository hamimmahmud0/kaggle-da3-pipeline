    "def _is_video_path(path: Path) -> bool:\n",
    "    # Prefer extension check; fall back to lightweight header sniffing.\n",
    "    if path.suffix.lower() in VIDEO_EXTENSIONS:\n",
    "        return True\n",
    "\n",
    "    try:\n",
    "        with open(path, \"rb\") as f:\n",
    "            head = f.read(64)\n",
    "    except Exception:\n",
    "        return False\n",
    "\n",
    "    # MP4/MOV/3GP family: ....ftyp....\n",
    "    if len(head) >= 8 and head[4:8] == b\"ftyp\":\n",
    "        return True\n",
    "    # Matroska/WebM\n",
    "    if head.startswith(bytes.fromhex(\"1a45dfa3\")):\n",
    "        return True\n",
    "    # AVI\n",
    "    if len(head) >= 12 and head.startswith(b\"RIFF\") and head[8:12] in {b\"AVI \", b\"AVIX\"}:\n",
    "        return True\n",
    "    # FLV\n",
    "    if head.startswith(b\"FLV\"):\n",
    "        return True\n",
    "    # Ogg container (may include video)\n",
    "    if head.startswith(b\"OggS\"):\n",
    "        return True\n",
    "    # ASF (WMV)\n",
    "    if head.startswith(bytes.fromhex(\"3026b2758e66cf11a6d900aa0062ce6c\")):\n",
    "        return True\n",
    "    # MPEG program/elementary stream (very rough)\n",
    "    if head[:4] in {bytes.fromhex(\"000001ba\"), bytes.fromhex(\"000001b3\")}:\n",
    "        return True\n",
    "\n",
    "    return False\n",

    "def _opencv_extract_frame(video_path: Path, frame_idx: int, out_path: Path):\n",
    "    cap = cv2.VideoCapture(str(video_path))\n",
    "    if not cap.isOpened():\n",
    "        cap.release()\n",
    "        raise RuntimeError('OpenCV failed to open video')\n",
    "\n",
    "    try:\n",
    "        frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT) or 0)\n",
    "        if frame_count <= 0:\n",
    "            frame_count = None\n",
    "\n",
    "        cap.set(cv2.CAP_PROP_POS_FRAMES, float(frame_idx))\n",
    "        ok, frame = cap.read()\n",
    "    finally:\n",
    "        cap.release()\n",
    "\n",
    "    if not ok or frame is None:\n",
    "        return False, frame_count\n",
    "\n",
    "    out_path.parent.mkdir(parents=True, exist_ok=True)\n",
    "    ok2 = cv2.imwrite(str(out_path), frame, [cv2.IMWRITE_PNG_COMPRESSION, 3])\n",
    "    if not ok2:\n",
    "        raise RuntimeError('Failed to write frame image')\n",
    "    return True, frame_count\n",
    "\n",

    "def _next_frame_task_for_video(video_key: str):\n",
    "    with FRAME_VIDEO_STATE_LOCK:\n",
    "        state = FRAME_VIDEO_STATE.get(video_key)\n",
    "        if not state:\n",
    "            return None\n",
    "        video_path = Path(state.get('video_path', ''))\n",
    "        video_stored_name = state.get('video_stored_name', '') or video_path.name\n",
    "        frames_dir = Path(state.get('frames_dir', ''))\n",
    "        next_idx = int(state.get('next_frame_idx', 0))\n",
    "        frame_count = state.get('frame_count', None)\n",
    "\n",
    "    if not video_path.exists():\n",
    "        raise FileNotFoundError(f'Video not found: {video_path}')\n",
    "\n",
    "    if frame_count is not None and next_idx >= int(frame_count):\n",
    "        return None\n",
    "\n",
    "    out_path = frames_dir / f'frame_{next_idx:06d}.png'\n",
    "    if not out_path.exists():\n",
    "        ok, fc = _opencv_extract_frame(video_path, next_idx, out_path)\n",
    "        if fc is not None:\n",
    "            frame_count = int(fc)\n",
    "        if not ok:\n",
    "            with FRAME_VIDEO_STATE_LOCK:\n",
    "                st = FRAME_VIDEO_STATE.get(video_key)\n",
    "                if st is not None and frame_count is not None:\n",
    "                    st['frame_count'] = int(frame_count)\n",
    "                    st['done'] = True\n",
    "            return None\n",
    "\n",
    "    rel = out_path.relative_to(FRAMES_DIR).as_posix()\n",
    "    task = {\n",
    "        'video_file_url': f'/uploads/{video_stored_name}',\n",
    "        'frame_no': int(next_idx),\n",
    "        'frame_download_url': f'/frames/{rel}',\n",
    "        'frame_local_path': str(out_path.resolve()),\n",
    "    }\n",
    "\n",
    "    msg = None\n",
    "    with FRAME_VIDEO_STATE_LOCK:\n",
    "        st = FRAME_VIDEO_STATE.get(video_key)\n",
    "        if st is not None:\n",
    "            if frame_count is not None:\n",
    "                st['frame_count'] = int(frame_count)\n",
    "                if st.get('report_every', 100) == 100:\n",
    "                    rep = max(10, int(frame_count) // 20) if int(frame_count) else 100\n",
    "                    st['report_every'] = int(rep)\n",
    "                    st['next_report'] = int(rep)\n",
    "\n",
    "            st['next_frame_idx'] = int(next_idx) + 1\n",
    "            extracted = int(st['next_frame_idx'])\n",
    "            fc2 = st.get('frame_count', None)\n",
    "            rep = int(st.get('report_every', 100) or 100)\n",
    "            nxt = int(st.get('next_report', rep) or rep)\n",
    "\n",
    "            if extracted == 1:\n",
    "                msg = f'Frame queue: started {video_path.name}'\n",
    "            elif extracted >= nxt:\n",
    "                if fc2:\n",
    "                    pct = int((extracted / int(fc2)) * 100)\n",
    "                    msg = f'Frame queue: {video_path.name} {extracted}/{int(fc2)} ({pct}%)'\n",
    "                else:\n",
    "                    msg = f'Frame queue: {video_path.name} extracted {extracted}'\n",
    "                st['next_report'] = int(nxt) + int(rep)\n",
    "\n",
    "            if fc2 and extracted >= int(fc2):\n",
    "                st['done'] = True\n",
    "                msg = f'Frame queue: done {video_path.name} ({extracted}/{int(fc2)})'\n",
    "\n",
    "    if msg:\n",
    "        web_print(msg)\n",
    "\n",
    "    return task\n",
    "\n",

    "def _fill_frame_queue(max_new: int = 200) -> int:\n",
    "    try:\n",
    "        max_new = int(max_new)\n",
    "    except Exception:\n",
    "        max_new = 200\n",
    "    max_new = max(1, min(max_new, 2000))\n",
    "\n",
    "    filled = 0\n",
    "    with FRAME_QUEUE_FILL_LOCK:\n",
    "        while filled < max_new:\n",
    "            with FRAME_QUEUE_LOCK:\n",
    "                max_len = FRAME_QUEUE.maxlen or 0\n",
    "                if max_len and len(FRAME_QUEUE) >= max_len:\n",
    "                    break\n",
    "\n",
    "            with FRAME_VIDEO_STATE_LOCK:\n",
    "                if not FRAME_VIDEO_ORDER:\n",
    "                    break\n",
    "                key = FRAME_VIDEO_ORDER.popleft()\n",
    "\n",
    "            try:\n",
    "                task = _next_frame_task_for_video(key)\n",
    "            except Exception as exc:\n",
    "                with FRAME_VIDEO_STATE_LOCK:\n",
    "                    FRAME_VIDEO_STATE.pop(key, None)\n",
    "                web_print(f'Frame queue: dropped video due to error: {exc}')\n",
    "                continue\n",
    "\n",
    "            if not task:\n",
    "                with FRAME_VIDEO_STATE_LOCK:\n",
    "                    FRAME_VIDEO_STATE.pop(key, None)\n",
    "                continue\n",
    "\n",
    "            with FRAME_QUEUE_LOCK:\n",
    "                max_len = FRAME_QUEUE.maxlen or 0\n",
    "                if (not max_len) or len(FRAME_QUEUE) < max_len:\n",
    "                    FRAME_QUEUE.append(task)\n",
    "                    filled += 1\n",
    "\n",
    "            with FRAME_VIDEO_STATE_LOCK:\n",
    "                st = FRAME_VIDEO_STATE.get(key)\n",
    "                if st and not st.get('done', False):\n",
    "                    FRAME_VIDEO_ORDER.append(key)\n",
    "                else:\n",
    "                    FRAME_VIDEO_STATE.pop(key, None)\n",
    "\n",
    "    return int(filled)\n",
    "\n",

    "            web_print(f\"[Device {device_no}] Failed (code {process.returncode}): {file_name}\")\n",

    "def _da3_sub_worker(device_no):\n",
    "    \"\"\"Sub-worker for a specific device that processes frame tasks\"\"\"\n",
    "    web_print(f\"[Device {device_no}] Sub-worker started, waiting for DA3...\")\n",
    "    \n",
    "    # Wait until isDA3Set = True\n",
    "    while not _is_da3_set():\n",
    "        if DA3_PROCESS_STOP_EVENT.is_set():\n",
    "            return\n",
    "        DA3_PROCESS_START_EVENT.wait(timeout=1.0)\n",
    "    \n",
    "    web_print(f\"[Device {device_no}] DA3 ready, starting processing\")\n",
    "    \n",
    "    while True:\n",
    "        if DA3_PROCESS_STOP_EVENT.is_set():\n",
    "            web_print(f\"[Device {device_no}] Sub-worker stopping\")\n",
    "            break\n",
    "        \n",
    "        # Get a task from the frame queue\n",
    "        task = da3_task_queue_pop()\n",
    "        \n",
    "        if task is None:\n",
    "            # No task available, wait a bit\n",
    "            time.sleep(0.5)\n",
    "            continue\n",
    "        \n",
    "        # Process the task\n",
    "        _run_da3_subprocess(task, device_no)\n",

    "def _da3_process_worker_manager():\n",
    "    \"\"\"Main worker that manages sub-workers for each device\"\"\"\n",
    "    web_print(\"DA3 process worker manager started\")\n",
    "    \n",
    "    # Wait until isDA3Set = True\n",
    "    while not _is_da3_set():\n",
    "        if DA3_PROCESS_STOP_EVENT.is_set():\n",
    "            return\n",
    "        DA3_PROCESS_START_EVENT.wait(timeout=1.0)\n",
    "    \n",
    "    web_print(\"DA3 is set, spawning sub-workers\")\n",
    "    \n",
    "    while True:\n",
    "        if DA3_PROCESS_STOP_EVENT.is_set():\n",
    "            web_print(\"DA3 process worker manager stopping\")\n",
    "            break\n",
    "        \n",
    "        # Get current device count\n",
    "        device_count = get_device_count()\n",
    "        device_nos = list(range(device_count))\n",
    "        \n",
    "        with DA3_PROCESS_WORKERS_LOCK:\n",
    "            # Stop workers that exceed current device count\n",
    "            while len(DA3_PROCESS_WORKERS) > device_count:\n",
    "                worker = DA3_PROCESS_WORKERS.pop()\n",
    "                # Workers will stop via stop event\n",
    "            \n",
    "            # Start new workers if needed\n",
    "            while len(DA3_PROCESS_WORKERS) < device_count:\n",
    "                device_no = len(DA3_PROCESS_WORKERS)\n",
    "                worker = Thread(\n",
    "                    target=_da3_sub_worker,\n",
    "                    args=(device_no,),\n",
    "                    daemon=True,\n",
    "                    name=f'DA3-Sub-Worker-{device_no}'\n",
    "                )\n",
    "                worker.start()\n",
    "                DA3_PROCESS_WORKERS.append(worker)\n",
    "                web_print(f\"Started DA3 sub-worker for device {device_no}\")\n",
    "        \n",
    "        # Sleep and check for device count changes\n",
    "        time.sleep(2.0)\n",
    "\n",

    "    original_path = path\n",
    "    if path.suffix.lower() == \".dav\":\n",

    "        except Exception as exc:\n",

    "            return\n",
    "        web_print(f\"Converted DAV -> MP4: {path.name}\")\n",
    "        source = (source + \" (dav->mp4)\") if source else \"dav->mp4\"\n",
    "\n",
    "    if not _is_video_path(path):\n",
    "        web_print(f\"[enqueue_video] : {path} is not a video path\")\n",
    "        return\n",
    "\n",
    "    try:\n",
    "        size_bytes = path.stat().st_size\n",
    "    except Exception:\n",

    "        \"source\": str(source or \"\"),\n",
    "        \"original_stored_name\": original_path.name if original_path != path else \"\",\n",
    "        \"original_path\": str(original_path) if original_path != path else \"\",\n",
    "    }\n",
    "\n",
    "    dropped = None\n",
    "    with VIDEO_QUEUE_LOCK:\n",
    "        if VIDEO_UPLOAD_QUEUE.maxlen and len(VIDEO_UPLOAD_QUEUE) >= VIDEO_UPLOAD_QUEUE.maxlen:\n",

    "    web_print(f\"Enqueued video{src}: {item['stored_name']} ({item['size_mb']} MB)\")\n",
    "\n",
    "    try:\n",
    "        enqueue_da3_job(path, source=source)\n",
    "    except Exception as exc:\n",
    "        web_print(f\"FPS adjust enqueue failed for {path.name}: {exc}\")\n",
    "\n",

    "    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:\n",

    "def _pick_port(host: str, preferred: int = 5000) -> int:\n",
    "    if _is_port_free(host, preferred):\n",
    "        return preferred\n",
    "    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:\n",
    "        sock.bind((host, 0))\n",
